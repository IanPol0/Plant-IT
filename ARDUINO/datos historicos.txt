#include <list> 

int humedadHistorica[100];
int luzHistorica[100];
int fechaHistorica[100];

void JuntarDatosHistoricos()
{
   for (int i = 0; i < 100; i++)
   {
       if (humedadHistorica[i] == 0)
       {
           humedadHistorica[i] == humedad;
           luzHistorica[i] == luz;
           fechaHistorica[i] == fecha;           
       }
   }
}
